# configuration of the Panasonic TH60's Smarttrack 

[DTRACK]
TYPE 		= DTRACK
NAME		= DTRACK
PROTOCOL	= DTRACK_PROTOCOL
CONNECTIONS	= DTRACKCONTROL, DTRACKDATA
SENSORS		= HEAD, STICK, HAND
HISTORY		= 10
ATTACHONLY	= FALSE

[DTRACK_PROTOCOL]
NAME		= dtrack2

[HEAD]
NAME		= HEAD_SENSOR
TYPE		= BODY
RAWID		= 0

[STICK]
NAME		= STICK_SENSOR
TYPE		= STICK2
RAWID		= 0

[HAND]
NAME		= HAND_SENSOR
TYPE		= BODY
RAWID		= 1

[DTRACKCONTROL]
TYPE		= TCP
DRIVERROLE	= CONTROLCONNECTION
ADDRESS		= 10.0.1.222
PORT		= 50105
DIRECTION	= OUTGOING

[DTRACKDATA]
TYPE = UDP
DRIVERROLE 	= DATACONNECTION
ADDRESS		= 10.0.1.29
PORT    	= 5001
DIRECTION	= INCOMING
